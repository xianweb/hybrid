# 设置各页面跳转

    <script src="js/mui.min.js"></script>
    <link href="css/mui.min.css" rel="stylesheet"/>
    <script type="text/javascript" charset="utf-8">
	mui.init({
		subpages: [
			{
			  url:'home.html',
			  id:'home',
			  styles:{ top:'45px', bottom: "50px" }
			},
			{
			  url:'discover.html',
			  id:'discover',
			  styles:{ top:'45px', bottom: "50px" }
			},
			{
			  url:'feedback.html',
			  id:'feedback',
			  styles:{ top:'45px', bottom: "50px" }
			},
			{
			  url:'about.html',
			  id:'about',
			  styles:{ top:'45px', bottom: "50px" }
			},
		]
	});
	mui.plusReady(function(){
		
		mui('.mui-bar-tab').on('tap','#home',function(){
			plus.webview.getWebviewById('home').show();
		})
		
		mui('.mui-bar-tab').on('tap','#discover',function(){
			plus.webview.getWebviewById('discover').show();
		})
		
		mui('.mui-bar-tab').on('tap','#feedback',function(){
			plus.webview.getWebviewById('feedback').show();
		})
		
		mui('.mui-bar-tab').on('tap','#about',function(){
			plus.webview.getWebviewById('about').show();
		})
		
	})
    </script>
</head>
<body>

在index.html头部加入js跳转逻辑代码

注意

这里的页面跳转是使用原生页面跳转，所有你再内置浏览器中单击底部导航没有效果，必须要真机测试或者使用模拟器测试，才能跳转
